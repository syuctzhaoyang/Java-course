    package human;
    enum Status{add,pri,exit}
    public class TestSum {

    public TestSum() {
      Status status = Status.add;
      int sum = 0;
      int i = 1;
      while(status != Status.exit){
        if(status == Status.add){
          if(i <= 100){
            sum+=i;
            i++;
          }else{
            status = Status.pri;
          }
        }else if(status == Status.pri){
          System.out.println(sum);
          status = Status.exit;
        }
      }

    }

    public static void main(String[] args) {
      new TestSum();
    }

  }
